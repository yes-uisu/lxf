        printf("####DEMO device open fail####\n");

        return (-1);}

    printf("write %d bytes data to /dev/demo \n",MAX_LEN);

    showbuf(buf);

    write(fd,buf,MAX_LEN);         //系统write，向设备节点/dev/demo 写入数组内容

    printf("Read %d bytes data from /dev/demo \n",MAX_LEN);

}
